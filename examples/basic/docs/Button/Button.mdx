```jsx
import * as React from 'react';

interface State {
  count: number;
}

export class Container extends React.Component<{}, State> {
  state = {
    count: 0
  };
  handleClick = () => {
    this.setState({
      count: this.state.count + 1
    })
  };
  render(): JSX.Element {
    return(
      <div>
        {this.state.count}
        <button onClick={this.handleClick}>click me</button>
      </div>
    )
  }
}

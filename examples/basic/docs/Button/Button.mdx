```jsx
import * as React from 'react';
import { connect } from 'react-redux';
import * as selectProducts from '../../state/products/selectors';
import { fetchProducts } from '../../state/products/actions';
import { ProductsModel } from '../models/ProductsModel';

/*
 * Definicion de los `Props` y `State` por medio de interfaces
 */

interface Props {
  fetchProducts?: Function;
  isFetching: boolean;
  error: boolean;
  products: ProductsModel[];
}

interface State {
  // ...
}

class Container extends React.Component <Props, State> {
  
  componentDidMount() {
    this.props.fetchProducts();
  }
  // Tipado de retorno especificando el retorno `JSX.Element`
  render(): JSX.Element {

    // Usamos destructure para obtener las variables que usaremos
    const { products, isFetching, error } = this.props;
    /**
     * 1. Validar el `isFeching` por defecto es `true` y
     * mostrar el `PlaceholderShimmer`del componente
     */
    if (isFeching) {
      return <ProductListPlaceholder />
    }
    /**
     * 2. validar el error y mostrar un componente de 
     * error específico o relizar una acción
     */
    if (error) {
      return <Error />
    }
    /**
     * 3. Renderear el componente
     * En el caso de tener que manipular los props para
     * entregarselos (Hidratación) al componente usar una constante:
     * 
     * const props = {
     *    totalProducts: products.length
     *    // ...
     * }
     * 
     * return <ProductList {...props} />
     */
    return <ProductList products={products} />
  }
}

/**
 * Definción de `mapStateToProps`, `mapDispatchToProps`
 * por medio de Fat-arrow con el retorno explícito
 */

const mapStateToProps = (state: { products }) => ({
  products: selectProducts.getData(state),
  isFetching: selectProducts.getFetchingState(state),
  error: selectProducts.getErrorState(state)
});

const mapDispatchToProps = dipatch => ({
  fetchProducts: () => dipatch(fetchProducts())
});

/**
 * Exportación del Container como resultado del `HOC` `connect`
 */
export const ProductsContainer = connect(
  mapStateToProps,
  mapDispatchToProps
)(Container)
```

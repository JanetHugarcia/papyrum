import React from 'react';
import ReactDOM from 'react-dom';
import { BrowserRouter, Route, NavLink } from 'react-router-dom';
import db from 'db';
console.log('db', db);

//_IMPORTS_

const components = [_COMPONENTS_];

const App = () => (
    <BrowserRouter>
            <div className='wrapper'>
                <ul className='sidebar'>
                    {Object.keys(db.entries).map((entry, i) => (
                        <li key={i}>
                            <NavLink exact activeStyle={{color:'black'}} to={db.entries[entry].route }>
                                {db.entries[entry].name}
                            </NavLink>
                        </li>
                    ))}
                </ul>
                <article className='content'>
                    {Object.keys(db.entries).map((entry, i) => (
                        <Route key={i} exact path='/' path={db.entries[entry].route} component={components[i]} />
                    ))}
                </article>
        </div>
    </BrowserRouter>
);

const node = document.createElement('div');
document.body.appendChild(node);

ReactDOM.render(<App />, node);
